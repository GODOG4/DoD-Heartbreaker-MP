country_event = {
	
	id = 800110

	title = "EVTNAME800110" # Royal Wedding
	desc = "EVTDESC800110"
	picture = "VictoriaWedding"

	trigger = {
		war = no
		civilized = yes
		NOT = { 
			has_country_flag = royal_wedding
			has_country_flag = royal_heir_born
		}
		OR = {
			government = prussian_constitutionalism
			government = absolute_monarchy
			government = hms_government
		}
		any_neighbor_country = {
			war = no
			NOT = { truce_with = THIS }		
			civilized = yes
			OR = {
				government = prussian_constitutionalism
				government = absolute_monarchy
				government = hms_government
			}
			relation = {
				who = THIS
				value = 50
			}
		}
	}
	mean_time_to_happen = {
		months = 100
		modifier = {
			factor = 1.2
			average_militancy = 2
		}
		modifier = {
			factor = 1.2
			average_militancy = 3
		}
		modifier = {
			factor = 1.2
			average_militancy = 4
		}

			factor = 1.5
			average_militancy = 5
		}
		modifier = {
			factor = 0.9
			NOT = { average_militancy = 2 }
		}
	}
	option = {
		name = "800110.A"
		random_country = {
			limit = {
				war = no
				neighbour = THIS
				NOT = { truce_with = THIS }		
				civilized = yes
				OR = {
					government = prussian_constitutionalism
					government = absolute_monarchy
					government = hms_government
				}
				relation = {
					who = THIS
					value = 50
				}
			}
			relation = {
				who = THIS
				value = 50
			}
		}
		set_country_flag = royal_wedding
		aristocrats = {
			limit = { is_primary_culture = yes }
			ideology = {
				factor = 0.2
				value = reactionary
			}
			ideology = {
				factor = 0.1
				value = conservative
			}
			consciousness = -3
			militancy = -1
		}
		any_pop = {
			limit = { is_primary_culture = yes }
			scaled_consciousness = {
				ideology = reactionary
				factor = -4
			}
			scaled_consciousness = {
				ideology = conservative
				factor = -2
			}
		}
		ai_chance = {
			factor = 80
		}
	}
	
	option = {
		name = "800110.B"
